import java.io.*;

public class Exp1Server {
    public static void main(String[] args) throws Exception {
        ServerSocket server=new ServerSocket(5000);
        System.out.println("Server waiting to connect...");
        Socket socket=server.accept();

        BufferedReader in=new BufferedReader(
            new InputStreamReader(socket.getInputStream())
        );
        String msg=in.readLine();
        System.out.println("Client says:"+ msg);
        int ans=calculate(msg);

        PrintWriter out=new PrintWriter(socket.getOutputStream(), true);
        out.println(ans);

        socket.close();
        server.close();
    }

    public static int calculate(String msg) {
        int ans=0;
        String words[]=msg.split(" ");
        ans=Integer.parseInt(words[1])+Integer.parseInt(words[3]);
        return ans;
    }
}




import java.util.ArrayList;
import java.util.List;

public class Exp2RaymondTreeBased {

    static int holder[] = {1, 1, 2, 2};   // parent/holder of each process
    static boolean hasToken[] = {true, false, false, false};

    static void requestCS(int p) {

        System.out.println("Process " + p + " requests Critical Section");
        int originalRequester = p;
        List<Integer> path = new ArrayList<>();
        path.add(p);

        while (!hasToken[p - 1]) {

            int parent = holder[p - 1];

            System.out.println("Request sent from P" + p + " to P" + parent);
            path.add(parent);
            p = parent;
        }

        System.out.println("Token found at P" + p);

        // Pass the token back down the tree and update holder pointers
        for (int i = path.size() - 1; i > 0; i--) {
            int giver = path.get(i);
            int receiver = path.get(i - 1);
            
            System.out.println("Token passed from P" + giver + " to P" + receiver);
            hasToken[giver - 1] = false;
            hasToken[receiver - 1] = true;
            holder[giver - 1] = receiver; // Reversing the edge towards new token holder
        }
        holder[originalRequester - 1] = originalRequester; // New token holder points to itself

        System.out.println("Process " + originalRequester + " enters Critical Section\n");
    }

    public static void main(String[] args) {

        System.out.println("Initial Token Holder: Process 1\n");

        requestCS(3);
        requestCS(4);
    }
}



public class Exp3DistributedGlobalAveraging {
    public static void main(String[] args) {
        double nodes[]={10, 20, 30, 40, 50};
        int n=nodes.length;

        for(int round=0; round<=15; round++) {
            double nextState[]=new double[5];
            for(int i=0; i<nodes.length; i++) {
                int left=(i-1+n)%n;
                int right=(i+1)%n;
                nextState[i]=(nodes[left]+nodes[right]+nodes[i])/3.0;
            }

            nodes=nextState;
            if(round%3==0) {
                System.out.print("Round "+ round +" values:");
                for(double val:nodes)
                    System.out.printf("%.2f ", val);
                System.out.println();
            }
        }
    }
}




import java.util.*;

public class Exp4LoadBalancing {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);

        System.out.print("Enter number of nodes:");
        int n=sc.nextInt();
        System.out.print("Enter number of processes:");
        int p=sc.nextInt();

        int load[]=new int[n];

        for(int i=0; i<p; i++) {
            int minNode=0;

            for(int j=1; j<n; j++) {
                if(load[j]<load[minNode])
                    minNode=j;
            }

            load[minNode]++;
            System.out.println("Process "+ (i+1) +" is assigned to Node "+ (minNode+1));
        }

        System.out.println(Arrays.toString(load));

        sc.close();
    }
}




import java.util.*;

public class Exp5Bully {
    static int processes=5;
    static boolean[] active={true, true, true, true, true};
    static int[] processId={1, 2, 3, 4, 5};

    public static void election(int c) {
        System.out.println("Process "+ c +" starts the election");
        int ans=c-1;

        for(int i=ans+1; i<processId.length; i++) {
            if(active[i]==true && processId[i]>ans) {
                System.out.println("P"+ (ans+1) +" sends elections message to P"+ (i+1));
                ans=i;
                System.out.println("Process "+ (ans+1) +" starts the election");
            }
        }

        System.out.println("Process "+ (ans+1) +" becomes the co-ordinator of the system");
    }
    
    public static void main(String args[]) {
        Scanner sc=new Scanner(System.in);

        System.out.println("5 Processes are running, which process to fail?");
        int p=sc.nextInt();
        active[p-1]=false;
        System.out.println("Which process detects the failure?");
        int c=sc.nextInt();
        election(c);

        sc.close();
    }
}




import java.net.*;
import java.io.*;

public class Exp6GroupReceiver {
    public static void main(String[] args) {
        try {
            int port = 5000;
            InetAddress group = InetAddress.getByName("225.0.0.1");
            InetSocketAddress groupAddress = new InetSocketAddress(group, port);
            NetworkInterface networkInterface = NetworkInterface.getByInetAddress(InetAddress.getLocalHost());

            MulticastSocket socket = new MulticastSocket(port);
            
            socket.joinGroup(groupAddress, networkInterface);

            System.out.println("Joined group using modern API. Waiting...");

            byte[] buffer = new byte[256];
            DatagramPacket packet = new DatagramPacket(buffer, buffer.length);
            
            socket.receive(packet);
            String message = new String(packet.getData(), 0, packet.getLength());
            System.out.println("Received: " + message);
            socket.leaveGroup(groupAddress, networkInterface);
            socket.close();
            
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

public class Exp7RicartAgrawala {
    static int n=3;
    static int timestamp[]={5, 3, 7};

    public static void cs(int p) {
        int flag=0;

        for(int i=0; i<n; i++) {
            if(i==p)
                continue;

            System.out.println("P"+ (p+1) +" sends request to P"+ (i+1));
            if(timestamp[p]<timestamp[i]) {
                System.out.println("P"+ (i+1) +" sends OK to P"+ (p+1));
            }
            else {
                System.out.println("P"+ (i+1) +" DEFERS reply to P"+ (p+1));
                flag++;
            }
        }

        if(flag==0)
            System.out.println("Process "+ p +" can enter into Critical Section");
        else
            System.out.println("Process "+ p +" cannot enter into Critical Section & has to wait");
    }

    public static void main(String args[]) {
        int p=2;
        System.out.println("Process "+ (p+1) +" requests Critical Section");
        cs(p);
    }
}





public class Exp8Bankers {

    public static void main(String[] args) {

        int processes = 3;
        int resources = 3;

        int allocation[][] = {
                {0, 1, 0},
                {2, 0, 0},
                {3, 0, 2}
        };

        int max[][] = {
                {7, 5, 3},
                {3, 2, 2},
                {9, 0, 2}
        };

        int available[] = {3, 3, 2};

        int need[][] = new int[processes][resources];

        // Calculate Need
        for (int i = 0; i < processes; i++) {
            for (int j = 0; j < resources; j++) {
                need[i][j] = max[i][j] - allocation[i][j];
            }
        }

        boolean finished[] = new boolean[processes];
        int safeSeq[] = new int[processes];
        int work[] = available.clone();

        int count = 0;

        while (count < processes) {

            boolean found = false;

            for (int i = 0; i < processes; i++) {

                if (!finished[i]) {

                    int j;
                    for (j = 0; j < resources; j++) {
                        if (need[i][j] > work[j])
                            break;
                    }

                    if (j == resources) {

                        for (int k = 0; k < resources; k++) {
                            work[k] += allocation[i][k];
                        }

                        safeSeq[count++] = i;
                        finished[i] = true;
                        found = true;
                    }
                }
            }

            if (!found) {
                System.out.println("System is NOT in safe state");
                return;
            }
        }

        System.out.print("Safe Sequence: ");
        for (int i = 0; i < processes; i++) {
            System.out.print("P" + safeSeq[i] + " ");
        }
    }
}




import java.util.Scanner;

public class Exp9Berkley {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);

        System.out.print("Enter number of nodes(including master):");
        int n=sc.nextInt();
        int time[]=new int[n];
        for(int i=0; i<n; i++) {
            System.out.print("Enter time for node "+ i +":");
            time[i]=sc.nextInt();
        }

        int avg=0, sum=0;
        for(int t:time)
            sum+=t;
        avg=sum/n;

        System.out.println("\nAdjusted Times:-");
        for(int i=0; i<n; i++) {
            int offset=avg-time[i];
            time[i]+=offset;
            System.out.println("Node "+ i +" is adjusted by "+ offset +" -> "+ time[i]);
        }

        sc.close();
    }
}

    public void run() {

            }
        }
    }
}
